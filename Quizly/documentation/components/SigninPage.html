import { Component, OnInit } from '@angular/core';
import { MessageService } from 'src/app/services/message.service';
import { AuthService } from 'src/app/services/auth.service';

@Component({
  selector: 'app-signin',
  templateUrl: './signin.page.html',
  styleUrls: ['./signin.page.scss'],
})
export class SigninPage implements OnInit {

  name: string;
  email: string;

  password: string;
  passwordCheck: string;

  passwordLength = 8;

  constructor(private messageService: MessageService,
              private authService: AuthService) { }

  ngOnInit() {
  }

  submit(): void {
    if (!this.name || !this.password || this.password.length < this.passwordLength) {
      this.messageService.alert(`Alle Felder müssen ausgefüllt werden. Die minimale Länge des Passworts ist ${this.passwordLength}.`)
      return;
    }

    if (this.password !== this.passwordCheck) {
      return;
    }

    // TODO send sign in data to backend
    // if no other of that name exists and everything else worked as well, log in
    const loggedIn = true;
    if (loggedIn) {
      // TODO store user credentials for future use
      this.authService.signin();
    }
  }
}
